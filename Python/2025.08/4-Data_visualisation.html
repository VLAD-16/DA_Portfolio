import pandas as pd
import matplotlib.pyplot as plt
import seaborn as sns
from numpy import median

# Завантаження файлу CSV в DataFrame
try:
    #df = pd.read_csv('facebook_ads_data (2.0).csv')
    #Place_your_url_below ▼
    url='https://drive.google.com/file/d/1P51VI_-GaHs8OU4OyHPPWFx70_PY2yxN/view?usp=sharing'
    df = pd.read_csv('https://drive.usercontent.google.com/download?id={}&export=download&authuser=0&confirm=t'.format(url.split('/')[-2]))
    print("Файл 'facebook_ads_data (2.0).csv' успішно завантажено.")
except FileNotFoundError:
    print("Помилка: файл 'facebook_ads_data (2.0).csv' не знайдено.")
    exit()

# Очистка даних
# Надані дані мають проблеми з заголовками та форматом.
# Перейменуємо стовпці для зручності
df.columns = df.columns.str.replace('"', '').str.strip()

# Конвертуємо 'ad_date' в datetime формат для зручного фільтрування
df['ad_date'] = pd.to_datetime(df['ad_date'])

# Конвертуємо числові стовпці у відповідний тип даних, замінюючи пусті значення на 0
numeric_cols = ['total_spend', 'total_impressions', 'total_clicks', 'total_value', 'cpc', 'cpm', 'ctr', 'romi']
for col in numeric_cols:
    df[col] = pd.to_numeric(df[col], errors='coerce').fillna(0)
    
# Створення DataFrame для 2021 року
df_2021 = df[df['ad_date'].astype(str).str.contains('2021-')]

print("\nЗагальна інформація про DataFrame:")
print(df.info())

print("\nПерші 5 рядків DataFrame (після очистки):")
print(df.head())

campaign_data = df.groupby('campaign_name').agg({
    'total_spend': 'sum',
    'total_value': 'sum'
})

# Розрахунок загального ROMI для кожної кампанії
campaign_data['romi'] = campaign_data['total_value'] / campaign_data['total_spend']
campaign_data['romi'] = campaign_data['romi'].replace([float('inf'), -float('inf')], 0).fillna(0)
