from statsmodels.stats.proportion import proportions_ztest # Z-test для пропорцій
from scipy.stats import chi2_contingency, permutation_test # Chi2-тест, пермутаційний тест

alpha = 0.05  # рівень значущості

# Підрахунок конверсії
summary = data.groupby('test_group')['conversion'].agg(['sum','count']).rename(columns={'sum':'conversions','count':'n'})
summary['cr'] = summary['conversions'] / summary['n']

# proportions z-test (для перевірки різниці пропорцій)
z_stat, p_z = proportions_ztest(summary['conversions'], summary['n'])

# chi2 test (для незалежності груп і конверсії)
chi2, p_chi2, _, _ = chi2_contingency(pd.crosstab(data['test_group'], data['conversion']))

# permutation test (робастний тест для різниці середніх)
res = permutation_test(
    (data.loc[data['test_group']=='a','conversion'],
     data.loc[data['test_group']=='b','conversion']),
    lambda a,b: a.mean()-b.mean(),
    n_resamples=10000, alternative='two-sided'
)

# 4. T-test

# Вивід
print('\n=== Summary ===')
print(summary.to_string(float_format='%.4f'))

print('\n=== Statistical Tests ===')
print(f'Z-test: z= {z_stat:.3f}, p={p_z:.20f} → {"Reject H₀ - option B wins." if p_z < alpha else "Non significant outcome - H₀ not rejected"}')
print(f'Chi2-test: χ²= {chi2:.3f}, p={p_chi2:.20f}')
print(f'Permutation test: stat= {res.statistic:.3f}, p={res.pvalue:.6f}')
